def obf_decode(token: str) -> str:
    # token is base64 of reversed utf-8 bytes
    raw = base64.b64decode(token.encode("ascii"))
    return raw.decode("utf-8")[::-1]

def validate_scale_320x240(out_path: str) -> Tuple[bool, str]:
    try:
        probe = ffprobe_json(out_path)
        v = get_first_stream(probe, "video")
        w = int(v.get("width", -1))
        h = int(v.get("height", -1))
        if w == 320 and h == 240:
            return True, "ok"
        return False, f"expected 320x240, got {w}x{h}"
    except Exception as e:
        return False, str(e)


def validate_gray_pixfmt(out_path: str) -> Tuple[bool, str]:
    try:
        probe = ffprobe_json(out_path)
        v = get_first_stream(probe, "video")
        pix = v.get("pix_fmt")
        if pix == "gray":
            return True, "ok"
        return False, f"expected pix_fmt=gray, got {pix}"
    except Exception as e:
        return False, str(e)
