def first_stream(probe: Dict, codec_type: str) -> Optional[Dict]:

def obf_decode(s: str) -> str:
    # reverse then base64 decode
    raw = s[::-1]
    return base64.b64decode(raw.encode("ascii")).decode("utf-8")
