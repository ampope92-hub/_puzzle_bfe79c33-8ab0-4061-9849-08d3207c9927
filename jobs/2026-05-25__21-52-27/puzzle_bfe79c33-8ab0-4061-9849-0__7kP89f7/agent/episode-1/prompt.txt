def parse_duration_seconds(probe: Dict) -> Optional[float]:
    # Prefer format.duration, fall back to stream.duration
    fmt = probe.get("format", {})
    dur = fmt.get("duration")
    if dur is not None:
        try:
            return float(dur)
        except ValueError:
            pass
    # fall back to first audio/video stream duration if available
    for s in probe.get("streams", []):
        if "duration" in s and s.get("duration") is not None:
            try:
                return float(s["duration"])
            except ValueError:
                continue
    return None


def run_ffmpeg_filter(input_path: str, output_path: str, filtergraph: str, *, is_audio: bool, extra_output_args: Optional[List[str]] = None) -> Tuple[bool, str]
:
    # Use -filter_complex to allow general filtergraphs; keep mapping simple.
    # For video: map v:0; For audio: map a:0
    extra_output_args = extra_output_args or []

    cmd = ["ffmpeg", "-y", "-v", "error", "-i", input_path, "-filter_complex", filtergraph]
    if is_audio:
        cmd += ["-map", "[a]" if "[a]" in filtergraph else "0:a:0"]
    else:
        cmd += ["-map", "[v]" if "[v]" in filtergraph else "0:v:0"]
    cmd += extra_output_args
    cmd += [output_path]

    cp = run(cmd)
    if cp.returncode != 0:
        return False, cp.stderr.strip() or cp.stdout.strip()
    return True, ""

def gen_level1(path: str) -> None:
    # 640x480 synthetic video, 2 seconds
    run([
        "ffmpeg", "-y", "-v", "error",
        "-f", "lavfi", "-i", "testsrc2=size=640x480:rate=30",
        "-t", "2",
        "-pix_fmt", "yuv420p",
        path,
    ])


def gen_level2(path: str) -> None:
    # Use AVI container, rawvideo, fixed pix_fmt to make gray validation strict.
    run([
        "ffmpeg", "-y", "-v", "error",
        "-f", "lavfi", "-i", "testsrc2=size=640x480:rate=30",
        "-t", "2",
        "-c:v", "rawvideo",
        "-pix_fmt", "yuv420p",
        path,
    ])
