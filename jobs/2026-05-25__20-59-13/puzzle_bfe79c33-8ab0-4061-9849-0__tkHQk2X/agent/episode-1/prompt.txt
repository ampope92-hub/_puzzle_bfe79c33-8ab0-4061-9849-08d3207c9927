def parse_duration_seconds(probe: Dict) -> Optional[float]:
    # Prefer format.duration; fall back to stream.duration
    dur = None
    fmt = probe.get("format", {})
    if "duration" in fmt:
        try:
            dur = float(fmt["duration"])
        except Exception:
            dur = None
    if dur is None:
        try:
            a = get_first_stream(probe, "audio")
            if "duration" in a:
                dur = float(a["duration"])
        except Exception:
            pass
    return dur
