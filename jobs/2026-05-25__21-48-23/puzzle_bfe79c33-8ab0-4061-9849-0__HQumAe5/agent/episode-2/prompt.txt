def run(cmd: List[str]) -> subprocess.CompletedProcess:
    return subprocess.run(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True)


def ensure_dirs() -> None:
    os.makedirs(ASSETS_DIR, exist_ok=True)
    os.makedirs(OUTPUT_DIR, exist_ok=True)


def ffprobe_json(path: str) -> Dict[str, Any]:
    cmd = [
        "ffprobe",
        "-v",
        "error",
        "-print_format",
        "json",
        "-show_format",
        "-show_streams",
        path,
    ]
    p = run(cmd)
    if p.returncode != 0:
        raise RuntimeError(f"ffprobe failed for {path}:\n{p.stderr}")
    return json.loads(p.stdout)


def get_first_stream(probe: Dict[str, Any], codec_type: str) -> Dict[str, Any]:
    for s in probe.get("streams", []):
        if s.get("codec_type") == codec_type:
            return s
    raise KeyError(f"No {codec_type} stream found")


def parse_duration_seconds(probe: Dict[str, Any]) -> Optional[float]:
    # Prefer stream duration if present; fallback to container duration.
    try:
        fmt = probe.get("format", {})
        if "duration" in fmt and fmt["duration"] not in (None, "N/A", ""):
            return float(fmt["duration"])
    except Exception:
        pass
    return None


def within(x: float, lo: float, hi: float) -> bool:
    return lo <= x <= hi


# --- solution obfuscation helpers (avoid plaintext filtergraphs in source) ---

def _rot13(s: str) -> str:
    out = []
    for ch in s:
        o = ord(ch)
        if 65 <= o <= 90:
            out.append(chr(((o - 65 + 13) % 26) + 65))
        elif 97 <= o <= 122:
            out.append(chr(((o - 97 + 13) % 26) + 97))
        else:
            out.append(ch)
    return "".join(out)


def decode_solution(token: str) -> str:
    # token: base64( rot13(solution)[::-1] )
    raw = base64.b64decode(token.encode("ascii")).decode("utf-8")
    return _rot13(raw[::-1])
